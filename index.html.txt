        let scrollPos = 0;
        let curDay = null; // 當前查看第幾天（修復備註儲存bug）

        // 讀取深色模式設定
        function loadDarkMode() {
            const dark = localStorage.getItem('darkMode');
            if(dark === '1') document.body.classList.add('dark-mode');
        }
        function saveDarkMode() {
            const isDark = document.body.classList.contains('dark-mode');
            localStorage.setItem('darkMode', isDark ? '1' : '0');
        }

        // 渲染行程列表
        function render() {
            const el = document.getElementById('list-v'); 
            el.innerHTML = '';
            data.forEach(d => {
                const c = document.createElement('div');
                // 讀取該天是否完成
                const doneStatus = localStorage.getItem(`done_${d.d}`) === '1';
                c.className = 'card' + (doneStatus ? ' done' : '');
                c.onclick = () => { 
                    scrollPos = window.scrollY; 
                    openDay(d.d); 
                };
                c.innerHTML = `
                    <div class="card-h">
                        <span class="card-d">DAY ${d.d}</span>
                        <span>${d.dt}</span>
                    </div>
                    <div class="card-t">${d.t}</div>
                `;
                el.appendChild(c);
            });
        }

        // 打開單日詳情
        function openDay(id) {
            curDay = id;
            const d = data.find(i => i.d === id);
            // 切換頁面
            document.getElementById('list-v').classList.remove('active');
            document.getElementById('detail-v').classList.add('active');
            window.scrollTo(0,0);
            // 填入內容
            document.getElementById('det-content').innerHTML = `
                <h2 style="font-size:16px; margin:10px 0;">Day ${d.d} - ${d.t}</h2>
                <div class="h-box">🏨 住宿：<div style="margin-top:5px;">${d.h}</div></div>

            `;
            // 載入備註
            document.getElementById('notes').value = localStorage.getItem(`nt_${id}`) || '';
            // 載入完成勾選狀態
            document.getElementById('isDone').checked = localStorage.getItem(`done_${id}`) === '1';
        }

        // 儲存行程完成狀態
        function saveDayStatus() {
            if(!curDay) return;
            const checked = document.getElementById('isDone').checked;
            localStorage.setItem(`done_${curDay}`, checked ? '1' : '0');
        }

        // 儲存備註
        function saveNotes() { 
            if(!curDay) return;
            localStorage.setItem(`nt_${curDay}`, document.getElementById('notes').value); 
        }

        // 返回列表
        function goBack() { 
            document.getElementById('detail-v').classList.remove('active'); 
            document.getElementById('list-v').classList.add('active'); 
            render(); // 重新渲染，更新完成標記樣式
            setTimeout(() => window.scrollTo(0, scrollPos), 50); 
        }

            if(page === 'list') {
                goBack();
